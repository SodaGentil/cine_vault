## 顺序表
#### 定义和初始化

##### 静态分配
```
//定义
#define MaxSize 50
typedef struct {

 Elemtype data[MaxSize];
 int length;              //线性表当前长度
}SqList;
//初始化
void InitList(SqList &L){

 L.length=0;
}
```
##### 动态分配
```
//定义
#define InitSize 100
typedef struct{

 Elemtype *data;
 int MaxSize,length;
}SeqList;
//初始化
void InitList(SeqList &L){

 L.data=(ElemType *)malloc(InitSize*sizeof(ElemType));
 L.length=0;
 L.MaxSize=InitSize;
}
```

#### 插入
```
                      插入位置：第几个  插入的元素
bool ListInsert (SqList &L, int i , ElemType e){
	
 if(i<1 || i>L.length+1)    //判断i的范围是否有效
	reture false;
 if(L.length>=MaxSize)     //存储空间满了
	return false;

 for(int j=L.length;j>i;j--)  //将第i个元素及之后的元素后移
	L.data[j]=L.data[j-1];
	
 L.data[i-1]=e;
 L.length++;
 return true;
}
```
